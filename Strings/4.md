```python
def f(a,b):
    flag=0
    for i in a:
        for j in b:
            if a.count(i) == b.count(j):
                flag=1
            else:
                flag=0
        return flag

            
a=input()
b=input()
flag=f(a,b)
if flag==1:
    print("Amagram")
    
else:
    print("Not a Amagram")
